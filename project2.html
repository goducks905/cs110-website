Project 2: Turtle Graphics Scene Using Functions

In Project 2, I expanded my scene by using provided drawing functions such as draw_rectangle, draw_triangle, draw_circle, and jump_to. I added trees, clouds, a mailbox, and improved the layout of the scene.

import turtle
import math

def setup_turtle():
    t = turtle.Turtle()
    t.speed(0)
    screen = turtle.Screen()
    screen.title("Turtle Graphics Assignment")
    return t, screen

def draw_rectangle(t, width, height, fill_color=None):
    if fill_color:
        t.fillcolor(fill_color)
        t.begin_fill()
    for _ in range(2):
        t.forward(width)
        t.right(90)
        t.forward(height)
        t.right(90)
    if fill_color:
        t.end_fill()

def draw_square(t, size, fill_color=None):
    if fill_color:
        t.fillcolor(fill_color)
        t.begin_fill()
    for _ in range(4):
        t.forward(size)
        t.right(90)
    if fill_color:
        t.end_fill()

def draw_triangle(t, size, fill_color=None):
    if fill_color:
        t.fillcolor(fill_color)
        t.begin_fill()
    for _ in range(3):
        t.forward(size)
        t.left(120)
    if fill_color:
        t.end_fill()

def draw_circle(t, radius, fill_color=None):
    if fill_color:
        t.fillcolor(fill_color)
        t.begin_fill()
    t.circle(radius)
    if fill_color:
        t.end_fill()

def jump_to(t, x, y):
    t.penup()
    t.goto(x, y)
    t.pendown()

def draw_scene(t):
    screen = t.getscreen()
    screen.bgcolor("skyblue")

    jump_to(t, -200, -200)
    draw_square(t, 100, fill_color="burlywood4")

    jump_to(t, -200, -200)
    draw_triangle(t, 100, fill_color="red")

    jump_to(t, 150, 150)
    draw_circle(t, 50, fill_color="yellow")

    jump_to(t, -175, -250)
    draw_rectangle(t, 30, 50, fill_color="maroon")

    jump_to(t, 130, -200)
    draw_rectangle(t, 40, 100, fill_color="sienna")
    jump_to(t, 150, -220)
    draw_circle(t, 50, fill_color="green")

    jump_to(t, -200, 200)
    draw_circle(t, 55, fill_color="white")

def main():
    t, screen = setup_turtle()
    draw_scene(t)
    screen.mainloop()

if __name__ == "__main__":
    main()
